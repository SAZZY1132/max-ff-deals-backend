const express = require('express'); const app = express(); const cors = require('cors'); const stripe = require('stripe')(process.env.STRIPE_SECRET_KEY); // Use from .env const bodyParser = require('body-parser'); app.use(cors()); app.use(express.json()); app.use(bodyParser.json()); // Sample product list (you can replace this with dynamic data) const PRODUCTS = { diamond_100: { name: "100 Diamonds", price: 1.99 }, diamond_310: { name: "310 Diamonds", price: 5.49 }, diamond_520: { name: "520 Diamonds", price: 8.99 }, }; app.post('/create-checkout-session', async (req, res) => { const { product_id } = req.body; const product = PRODUCTS[product_id]; if (!product) return res.status(400).json({ error: "Invalid product." }); try { const session = await stripe.checkout.sessions.create({ payment_method_types: ['card'], mode: 'payment', line_items: [ { price_data: { currency: 'usd', product_data: { name: product.name, }, unit_amount: Math.round(product.price * 100), // In cents }, quantity: 1, }, ], success_url: 'https://max-ff-deals.vercel.app/success.html', cancel_url: 'https://max-ff-deals.vercel.app/cancel.html', }); res.json({ url: session.url }); } catch (err) { console.error("Stripe Error:", err); res.status(500).json({ error: 'Something went wrong with payment.' }); } }); app.get('/', (req, res) => { res.send("MAX FF DEALS Backend is alive!"); }); const PORT = process.env.PORT || 5000; app.listen(PORT, () => { console.log(`Server running on port ${PORT}`); });
